  READ staffCount
  PROMPT "Enter number of children:"
  READ childCount

  CREATE arrays:
    Staff[staffCount]
    Children[childCount]
    Ratings[parentCount][staffCount]

  FOR each staff member i from 0 to staffCount - 1 DO
    PROMPT for name, birth year, gender, job title, weekly hours, wage rate, hiring year
    INITIALIZE Staff[i] with entered values
  END FOR

    INITIALIZE Children[j] with entered values
  END FOR

  FOR each parent p from 0 to parentCount - 1 DO
    FOR each staff member s from 0 to staffCount - 1 DO
      PROMPT "Rate staff s:"
      READ Ratings[p][s]
    END FOR
  END FOR

  // Calculate average ratings for each staff
  FOR s = 0 to staffCount - 1 DO
    sum = 0
    FOR p = 0 to parentCount - 1 DO
      sum = sum + Ratings[p][s]
    END FOR
    Staff[s].avgRating = sum / parentCount
  END FOR

  // Sort staff by avgRating (descending)
  FOR i = 0 to staffCount - 2 DO
    FOR j = 0 to staffCount - i - 2 DO
      IF Staff[j].avgRating < Staff[j+1].avgRating THEN
        SWAP Staff[j] and Staff[j+1]
      END IF
    END FOR
  END FOR
